# New Exchange-Aware Holiday Import Commands
# Use these instead of the old "holidays import" commands

# 1. Import all US Equity holidays (2024-2026) - RECOMMENDED
time import-holidays data/holidays/us_equity/us_equity_2024-2026.json

# 2. Import from old CSV files (backward compatible)
# Note: Add --exchange parameter to specify which exchanges get the holidays
time import-holidays data/holidays/2015_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2016_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2017_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2018_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2019_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2020_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2021_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2022_Holiday_Schedule.csv --exchange US_EQUITY
time import-holidays data/holidays/2023_Holiday_Schedule.csv --exchange US_EQUITY

# NOTE: 2024-2026 are already included in the JSON file above!

# 3. Dry run (validate before importing)
time import-holidays data/holidays/us_equity/us_equity_2024-2026.json --dry-run

# 4. List available exchange groups
time list-groups

# 5. Query holidays by exchange
time holidays --exchange NYSE --year 2024
time holidays --exchange NASDAQ --year 2024

# Exchange Groups Available:
# - US_EQUITY: NYSE, NASDAQ, AMEX, NYSE_ARCA
# - US_OPTIONS: CBOE, ISE, PHLX
# - US_FUTURES: CME, CBOT, NYMEX, COMEX
# - UK_EQUITY: LSE, AIM
# - JP_EQUITY: TSE, JPX
# - EU_EQUITY: XETRA, EURONEXT
